# Config File Manager

```js
__.FileManager.config = {
    enable_select_folder: true,
    enable_upload: true,
    enable_create_folder: true,
    enable_delete: true,
    enable_rename: true,
    enable_download: true,
    enable_view: true,
    extensions: ["img", "jpeg", "jpg", "png", "gif", "svg", "webp"],
}
```
